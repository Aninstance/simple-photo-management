# Examples use:
  - database: spm
  - username: picard
  - password: mYPas$worD

# access postgres
$ psql --host=localhost --dbname=spm --username=picard

# dump sql
$ env PGPASSWORD="mYPas$worD" pg_dump -h localhost -U picard spm > spm.sql
Note: to drop & recreate tables on restore, add the --clean flag

# restore sql
$ env PGPASSWORD="mYPas$worD" psql -h localhost -U picard spm < spm.sql
Note, if using Docker:
$ docker exec -i my_db_container psql -U picard spm < spm.sql
